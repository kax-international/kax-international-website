const CSV_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vQ2rFzA-udsqKKh5FHf9-naLtllz9qpM6cm9JiugzGLh9FrGSnGbKrfxX0Puf0lzSBI5dxVMTm91Ndn/pub?output=csv"; const gallery = document.getElementById("gallery"); const searchInput = document.getElementById("searchInput"); let allData = []; async function loadData() { try { const response = await fetch(CSV_URL); const csvText = await response.text(); const rows = csvText .split("\n") .map(row => row.trim()) .filter(row => row); const headers = parseCSVRow(rows[0]); allData = rows.slice(1).map(row => { const cols = parseCSVRow(row); return { timestamp: cols[0] || "", yourName: cols[1] || "", assignmentName: cols[2] || "", mail: cols[3] || "", numbering: cols[4] || "", playType: cols[5] || "", keywords: cols[6] || "", image: convertDriveUrl(cols[7] || "") }; }); render(allData); } catch(error) { console.error(error); gallery.innerHTML = `
Failed to load spreadsheet.
`; } } function parseCSVRow(row) { const result = []; let current = ""; let insideQuotes = false; for(let i = 0; i < row.length; i++) { const char = row[i]; if(char === '"') { insideQuotes = !insideQuotes; } else if(char === "," && !insideQuotes) { result.push(current); current = ""; } else { current += char; } } result.push(current); return result; } function convertDriveUrl(url) { if(!url) return ""; const match = url.match(/\/d\/(.*?)\//); if(match && match[1]) { return `https://drive.google.com/thumbnail?id=${match[1]}&sz=w1000`; } return url; } function render(data) { gallery.innerHTML = ""; data.forEach(item => { const card = document.createElement("div"); card.className = "card"; card.innerHTML = ` ${ item.image ? `` : "" }

${item.assignmentName}

Name: ${item.yourName}

Play Type: ${item.playType}

Keywords: ${item.keywords}

Number: ${item.numbering}

`; gallery.appendChild(card); }); } searchInput.addEventListener("input", () => { const keyword = searchInput.value.toLowerCase(); const sorted = [...allData].sort((a, b) => { const aScore = ( a.assignmentName.toLowerCase().includes(keyword) || a.keywords.toLowerCase().includes(keyword) || a.playType.toLowerCase().includes(keyword) ) ? 1 : 0; const bScore = ( b.assignmentName.toLowerCase().includes(keyword) || b.keywords.toLowerCase().includes(keyword) || b.playType.toLowerCase().includes(keyword) ) ? 1 : 0; return bScore - aScore; }); render(sorted); }); loadData();
